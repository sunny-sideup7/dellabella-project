      $.fn.digits = function () {
        return this.each(function () {
          $(this).text(
            $(this)
              .text()
              .replace(/(\d)(?=(\d\d\d)+(?!\d))/g, "$1,")
          );
        });
      };
      $(document).ready(function () {
        var instock = { JKT: 0, MDN: 0, BDG: 0, SBY: 0 };

        $(".changePhoto").click(function (e) {
          e.preventDefault();
          $("#mainPhoto").attr("src", $(this).attr("src"));
        });

        $("select[name=provinsi]").change(function (e) {
          // get city lists
          $.ajax({
            url:
              "https://wahanasuperstore.com/city/getByStateID/" + $(this).val(),
            cache: false,
            beforeSend: function () {
              $("select[name=kota]").html();
            },
            success: function (html) {
              $("select[name=kota]").html(html);
              $.ajax({
                url:
                  "https://wahanasuperstore.com/city/getShippingPriceById/" +
                  $("select[name=kota]").val(),
                cache: false,
                beforeSend: function () {},
                success: function (price) {
                  $("#estprice").html($("#productWeight").val() * price);
                  $("#estprice").digits();
                },
              });
            },
          });
        });

        $("select[name=kota]").change(function (e) {
          $.ajax({
            url:
              "https://wahanasuperstore.com/city/getShippingPriceById/" +
              $(this).val(),
            cache: false,
            beforeSend: function () {},
            success: function (price) {
              $("#estprice").html($("#productWeight").val() * price);
              $("#estprice").digits();
            },
          });
        });

        $.ajaxSetup({
          headers: {
            "X-CSRF-TOKEN": $('meta[name="csrf-token"]').attr("content"),
          },
        });

        var timer;

        $("select[name=kotaAsal]").change(function (e) {
          if (instock[$(this).val()] == 1) {
            $("select[name=jumlahKirim").removeAttr("disabled");
            $(".btn .add_to_cart").removeAttr("disabled");
            $(".btn .add_to_checkout").removeAttr("disabled");
            $("#ada").show();
            $("#kosong").hide();
          } else {
            $("select[name=jumlahKirim").attr("disabled", "true");
            $(".btn .add_to_cart").attr("disabled", "true");
            $(".btn .add_to_checkout").attr("disabled", "true");
            $("#kosong").show();
            $("#ada").hide();
          }
        });

        $(".btn.add_to_checkout:not([disabled])").click(function (e) {
          e.preventDefault();
          clearTimeout(timer);
          timer = setTimeout(function () {
            if ($("#formCart select[name='qty']").val() != "") {
              $.ajax({
                url: "https://dellabellaphotography.com/news/",
                type: "POST",
                data: $("#formCart").serialize(),
                beforeSend: function () {
                  $("#alertbawah").hide();
                },
                success: function (data) {
                  $("#alertbawah").show();
                  $(".wscarticon .roundpoint").html(data);
                },
                error: function (jqXHR, textStatus, errorThrown) {},
              });
            } else {
              $(".btn.add_to_cart").click();
            }
          }, 500);
        });
      });
